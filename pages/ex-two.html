Tính và hiển thị khoảng cách từ hai điểm (dùng Popup hiển thị tại vị trí điểm cuối của Polyline): 


var startPoint; 
var endPoint;
var marker;
var polyline;

map.on('click', (e) => {
  if (!startPoint) {
    startPoint = e.latlng;
    marker = L.marker(startPoint).addTo(map);
  } else if (!endPoint) {
    endPoint = e.latlng;
    marker = L.marker(endPoint).addTo(map);
    marker.bindPopup(`Khoảng cách ${startPoint.distanceTo(endPoint).toFixed(2)} km`).openPopup();
  }

  if (startPoint && endPoint) {
    polyline = L.polyline([startPoint, endPoint], {color: 'red'}).addTo(map);
  }
})
