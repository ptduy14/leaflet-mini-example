Vẽ Polyline bằng thao tác click chuột trên bản đồ. Trong đó:
- Click chuột: tạo marker và điểm mới đưa vào Polyline
- Kết thúc vẽ bằng thao tác double click 


var points = [];
var marker;
var polyline;
      
// khởi tạo hàm vẽ polyline
const drawPolygon = (e) => {
  points.push(e.latlng);
  marker = L.marker(e.latlng).addTo(map)
      
  if (points.length >= 2) {
    polyline = L.polyline(points, {color: 'red'}).addTo(map);
  }
}
      
// tạo sự kiện click để vẽ
map.on('click', drawPolygon)
      
// tạo hàm off sự kiện click
const endDrawing = () => {
  map.off('click', drawPolygon);
};
      
// khởi tạo event double click thì gọi hàm off sự kiện slick
map.on('dblclick', endDrawing);
